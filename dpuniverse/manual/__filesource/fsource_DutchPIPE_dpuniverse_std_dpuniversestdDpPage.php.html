  149.                 $direction[$y] = $direction[0];

  156.  
  157.         $newdirs = array();

  339.     function tell($data, &$from = NULL)
  340.     {

  345.  
  346.         if (func_num_args() <= 1)  {

  349.                     $ob->tell($data, $this);

  364.                 foreach ($inv as &$ob) {

  373.                             && ($tmp =
  374.                             $data[$ob->displayMode])) {

  384.                     if (FALSE === in_array($ob, $from, TRUE)) {
  385.                         $ob->tell($data, $this);
