  632.     /**

  636.      * This method is usually called from the reset method of pages and npc's.
  637.      *
  638.      * If $what is a string, searches for an object with that id. If $what is an
  639.      * object, searches for that object. Searches are done in the inventory of
  640.      * this object.
  641.      *
  642.      * A $number can be given to search for more objects instead of just one.
  643.      * For example, it could check for the presence of 5 roses, and if there
  644.      * are only 3, create another 2.

  646.      * If $mustBeHere is FALSE (it is TRUE by default), it will not replenish
  647.      * objects which are still in another part of the universe, instead of
  648.      * having been destroyed. For example, the 2 missing roses could have been
  649.      * taken by a user who is now on another page, and with this argument set
  650.      * to FALSE, no extra objects will be created.
  651.      *
  652.      * If $mustBeHere is TRUE, $moveHere can be set to TRUE (it is FALSE by
  653.      * default) to move objects which still exist but are located in another
  654.      * part of the universe, back to this object, its original location.
  655.      *
  656.      * @param      mixed     $what       string (id) or object to search for
  657.      * @param      int       $number     number of such objects to search for
  658.      * @param      boolean   $mustBeHere search only here for object?
  659.      * @param      boolean   $moveHere   replenish from objects elsewhere?
  660.      */

  662.             $moveHere = FALSE)
  663.     {

  665.         if (0 < count($this->mCheckPresentObjects)) {
  666.             $keys = array_keys($this->mCheckPresentObjects);
  667.             foreach ($keys as $key) {
  668.                 if (!isset($this->mCheckPresentObjects[$key])) {
  669.                     continue;
  670.                 }
  671.                 if (empty($this->mCheckPresentObjects[$key])
  672.                         || $this->mCheckPresentObjects[$key]->isRemoved) {
  673.                     unset($this->mCheckPresentObjects[$key]);
  674.                     continue;
  675.                 }
  676.                 $ob = $this->mCheckPresentObjects[$key];
  677.                 if ((is_object($what) && $ob === $what)
  678.                         || (is_string($what) && $ob->location === $what)) {
  679.                     if ($mustBeHere) {
  680.                         if (!$this->isPresent($ob)) {
  681.                             if ($moveHere) {
  682.                                 $ob->moveDpObject($this);
  683.                                 if (++$found === $number) {
  684.                                     return;
  685.                                 }
  686.                             }
  687.                         } elseif (++$found === $number) {
  688.                             return;
  689.                         }
  690.                     }
  691.                     elseif (++$found === $number) {
  692.                         return;
  693.                     }
  694.                 }
  695.             }
  696.         }
  697.         while ($found < $number) {
  698.             $ob = get_current_dpuniverse()->newDpObject($what);
  699.             $ob->moveDpObject($this);
  700.             $this->mCheckPresentObjects[] =& $ob;
  701.             $found++;
  702.         }
  703.     }
