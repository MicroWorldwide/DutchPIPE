  324.     /**
  325.      * Moves this object into the inventory of another object
  326.      *
  327.      * @param   mixed   &$target_ob path or object to move into to
  328.      * @param   boolean $simple     skip some checks
  329.      * @return  int     FALSE for success, an error code for failure
  330.      */
  331.     function moveDpObject(&$target_ob, $simple = FALSE)
  332.     {
  333.         /* Retrieve the target object, or fail */
  334.         if ((!is_string($target_ob) && !is_object($target_ob))
  335.                 || (is_string($target_ob)
  336.                 && !($target_ob =
  337.                     get_current_dpuniverse()->getDpObject($target_ob)))) {
  338.             return E_MOVEOBJECT_BADDEST;
  339.         }
  340.  
  341.         $curr_env = $this->getEnvironment();
  342.  
  343.         /* Do more checking? */
  344.         if (FALSE === $simple) {
  345.             /* Checks if the living carrying this object may not drop it: */
  346.             if (FALSE !== $curr_env && $curr_env->getProperty('is_living')
  347.                     && FALSE !== $curr_env->getProperty('no_drop')) {
  348.                 return E_MOVEOBJECT_NODROP;
  349.             }
  350.  
  351.             /* Checks if target is neither a page nor a living */
  352.             if (FALSE !== $target_ob->getEnvironment() &&
  353.                     FALSE === $target_ob->getProperty('is_living')) {
  354.                 /*
  355.                  * Checks if thie object may be inserted into containers, for
  356.                  * instance, drinks could disallow this, to prevent 'spilling':
  357.                  */
  358.                 if (method_exists($this, 'preventInsert')
  359.                         && (bool)$this->preventInsert()) {
  360.                     return E_MOVEOBJECT_NOSRCINS;
  361.                 }
  362.  
  363.                 /* Checks if the target is a container, or fail */
  364.                 if (FALSE === ((bool)$target_ob->getProperty('is_container'))) {
  365.                     return E_MOVEOBJECT_NODSTINS;
  366.                 }
  367.             }
  368.  
  369.             /* Checks if the object may be taken by livings */
  370.             if ($target_ob->getProperty('is_living')
  371.                     && (bool)$this->getProperty('no_take')) {
  372.                 return E_MOVEOBJECT_NOGET;
  373.             }
  374.         }
  375.  
  376.         /*
  377.          * If we're moving to a new environment, check for user defined
  378.          * functions to prevent movement
  379.          */
  380.         if ($curr_env !== $target_ob) {
  381.             if (method_exists($this, 'preventMove') && ($move_err =
  382.                     $this->preventMove($curr_env, $target_ob))) {
  383.                 return $move_err;
  384.             }
  385.             if (!is_null($curr_env)) {
  386.                 if (method_exists($this, 'preventLeave') && ($move_err =
  387.                         (int)$curr_env->preventLeave($this, $target_ob))) {
  388.                     return $move_err;
  389.                 }
  390.             }
  391.             if (method_exists($this, 'preventEnter') && ($move_err =
  392.                     (int)$target_ob->preventEnter($this, $curr_env))) {
  393.                 return $move_err;
  394.             }
  395.         }
  396.  
  397.         /* Move the object */
  398.         get_current_dpuniverse()->moveDpObject($this, $target_ob);
  399.  
  400.         /* Post-movement stuff */
  401.         if ($curr_env !== $target_ob) {
  402.             if (method_exists($this, 'event')) {
  403.                 $this->event(EVENT_CHANGED_ENV, $curr_env, $target_ob);
  404.             }
  405.             if (FALSE !== ($curr_env)) {
  406.                 $old_page = $curr_env;
  407.                 while (FALSE !== $old_page->getEnvironment()) {
  408.                     $old_page = $old_page->getEnvironment();
  409.                 }
  410.                 if (method_exists($curr_env, 'event')) {
  411.                     $curr_env->event(EVENT_LEFT_INV, $this, $target_ob);
  412.                 }
  413.             }
  414.             $new_page = $target_ob;
  415.             while (FALSE !== $new_page->getEnvironment()) {
  416.                 $new_page = $new_page->getEnvironment();
  417.             }
  418.  
  419.             $abstract_class = FALSE === $target_ob->getEnvironment()
  420.                 ? 'inventory' : 'inventory2';
  421.  
  422.             if (isset($old_page) && $old_page === $new_page) {
  423.                 $new_page->tell(array(
  424.                     'abstract' => '<moveDpElement id="'
  425.                         . $this->getUniqueId()
  426.                         . '" where="' . $target_ob->getUniqueId()
  427.                         . '" class="' . $abstract_class
  428.                         . '">&#160;</moveDpElement>'));
  429.                 if ($target_ob === $new_page) {
  430.                     $new_page->tell(array(
  431.                         'graphical' => '<addDpElement id="'
  432.                             . $this->getUniqueId() . '" where="'
  433.                             . $target_ob->getUniqueId()
  434.                             . '" class="title_img">'
  435.                             . $this->getAppearance(1, FALSE, $new_page,
  436.                             'graphical') . '</addDpElement>'));
