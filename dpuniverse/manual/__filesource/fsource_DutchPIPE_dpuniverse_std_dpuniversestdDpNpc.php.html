  117.     function tell($data)
  118.     {
  119.         if (empty($data)) {
  120.             return;
  121.         }
  122.  
  123.         if (is_array($data)) {
  124.             $data = $data[$this->getProperty('display_mode')];
  125.         }
  126.         if (strlen($data) >=3 && substr($data, 0, 1) == '<'
  127.                 && FALSE !== ($pos = strpos($data, '>'))) {
  128.             $type = substr($data, 1, $pos - 1);
  129.             $endpos = strrpos($data, '<');
  130.             $data = "<$type><![CDATA[" . substr($data, strlen($type) + 2,
  131.                 $endpos - strlen($type) - 2) . ']]>' . substr($data, $endpos);
  132.         } else {

  134.             $data = "<message><![CDATA[$data]]></message>";
  135.         }
  136.  
  137.         if (strlen($data) > 19
  138.                 && FALSE !== ($pos1 = strpos($data, '<location><![CDATA['))
  139.                 && FALSE !== ($pos2 = strpos($data, ']]></location>'))
  140.                 && $pos2 > $pos1 + 14) {
  141.             $data = substr($data, 0, $pos2);
  142.             $data = substr($data, $pos1 + 19);
  143.             $newlocation = $data;

  151.
