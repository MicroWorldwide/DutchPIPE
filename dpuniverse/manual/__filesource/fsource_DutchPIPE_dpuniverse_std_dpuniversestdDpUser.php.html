  164.     /**
  165.      * Sends something to dpclient-js.php running on the user's browser
  166.      *
  167.      * Sends the given data to the user's browser. It's up the user's DutchPIPE
  168.      * Javascript client, dpclient-js.php by default, what to do with the
  169.      * received content. dpclient-js.php can be send data like
  170.      * '<message>hello world</message>' and '<window>hello world</window>',
  171.      * to show something in the message area and pop up a window respectively.
  172.      *
  173.      * @param      string    $data      message string
  174.      */
  175.     function tell($data, &$from = NULL)
  176.     {
  177.         if ((FALSE === is_string($data) || 0 === strlen($data)) &&
  178.                 (FALSE === is_array($data)) || 0 === count($data)) {
  179.             return;
  180.         }
  181.  
  182.         if (FALSE === is_null($from)
  183.                 && (FALSE === ($env = $this->getEnvironment())
  184.                 || $env !== $from)) {
  185.             echo dptext("Message skipped, no longer in page\n");
  186.             return;
  187.         }
  188.  
  189.         // If this user is the same user doing the current HTTP request, tell
  190.         // straight away. Otherwise, store the message for the next time we get
  191.         // a HTTP request from the user.
  192.         if (TRUE === get_current_dpuniverse()->isNoDirectTell()
  193.                 || $this !== get_current_dpuser()) {
  194.             echo sprintf(dptext("Storing %s: %s\n"), $this->getTitle(),
  195.                 (strlen($data) > 512 ? substr($data, 0, 512) : $data));
  196.             get_current_dpuniverse()->storeTell($this, $data, $from);
  197.             return;
  198.         }
  199.         if (is_array($data)) {
  200.  
  201.             $data = $data[$this->getProperty('display_mode')];
  202.         }
  203.  
  204.         // Gets the message type from $data which holds a string with the format
  205.         // <type>tellstuff</type>. If no type was given in the $data string, we
  206.         // consider it type 'message':
  207.         list($mtype_start, $data, $mtype_end) = $this->_tellParseTag($data);
  208.  
  209.         $this->_tellDoTell($mtype_start, $data, $mtype_end);
  210.     }

  212.     private function _tellParseTag($data)
  213.     {
  214.          if (strlen($data) >=3 && substr($data, 0, 1) == '<'
  215.                 && FALSE !== ($pos = strpos($data, '>'))) {
  216.             $mtype_start = substr($data, 1, $pos - 1);
  217.             $endpos = strrpos($data, '<');
  218.             $mtype_end = substr($data, $endpos + 2, -1);
  219.  
  220.             $data = "<![CDATA[" . substr($data, strlen($mtype_start) + 2,
  221.                 $endpos - strlen($mtype_start) - 2) . ']]>';
  222.         } else {
  223.             $mtype_start = $mtype_end = 'message';
  224.             $data = "<![CDATA[$data]]>";
  225.         }
  226.  
  227.         return array($mtype_start, $data, $mtype_end);
  228.     }

  230.     private function _tellDoTell($mtype_start, $data, $mtype_end)
  231.     {
  232.         $universe = get_current_dpuniverse();
  233.  
  234.         if (isset($this->_GET['ajax'])) {
  235.             if ($mtype_end === 'message' && $data === '<![CDATA[empty]]>') {
  236.                 $mtype_start = $mtype_end = '';
  237.                 $data = '1';
  238.             }
  239.         }
  240.         if ($data !== '1' || !$universe->isToldSomething()) {
  241.             if ($mtype_start !== '') {
  242.                 $data = "<$mtype_start>$data</$mtype_end>";
  243.             }
  244.             if ($data !== '1') {
  245.                 echo sprintf(dptext("Telling %s: %s\n"), $this->getTitle(),
  246.                     (strlen($data) > 256 ? substr($data, 0, 256) : $data));
  247.             }
  248.             if ($mtype_end === '' || $mtype_end === 'header'
  249.                     || $mtype_end === 'cookie' || $mtype_end === 'location') {
  250.                 $universe->tellCurrentDpUserRequest($data);
  251.             } else {
  252.                 $universe->tellCurrentDpUserRequest('<event count="'
  253.                     . $this->mEventCount++ . '" time="' . time() . '">' . $data
  254.                     . '</event>');
  255.             }
  256.         }
  257.     }
