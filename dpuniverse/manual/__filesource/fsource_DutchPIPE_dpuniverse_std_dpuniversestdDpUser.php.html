  291.     /**
  292.      * Sends something to dpclient-js.php running on the user's browser
  293.      *
  294.      * Sends the given data to the user's browser. It's up the user's DutchPIPE
  295.      * Javascript client, dpclient-js.php by default, what to do with the
  296.      * received content. dpclient-js.php can be send data like
  297.      * '<message>hello world</message>' and '<window>hello world</window>',
  298.      * to show something in the message area and pop up a window respectively.
  299.      *
  300.      * Can be binded to an environment. This is useful for locations producing
  301.      * messages: the user could be moving to another location before the client
  302.      * catches the new message with AJAX, in which case the user should not get
  303.      * the message.
  304.      *
  305.      * @param      string    $data         message string
  306.      * @param      object    &$binded_env  optional binded environment
  307.      */
  308.     function tell($data, &$binded_env = NULL)
  309.     {
  310.         if ((FALSE === is_string($data) || 0 === strlen($data)) &&
  311.                 (FALSE === is_array($data)) || 0 === count($data)) {
  312.             return;
  313.         }
  314.  
  315.         if (FALSE === is_null($binded_env)
  316.                 && (FALSE === ($env = $this->getEnvironment())
  317.                 || $env !== $binded_env)) {
  318.             echo dptext("Message skipped, no longer in page\n");
  319.             return;
  320.         }
  321.  
  322.         // If this user is the same user doing the current HTTP request, tell
  323.         // straight away. Otherwise, store the message for the next time we get
  324.         // a HTTP request from the user.
  325.         if (TRUE === get_current_dpuniverse()->isNoDirectTell()
  326.                 || $this !== get_current_dpuser()) {
  327.             //echo sprintf(dptext("Storing %s: %s\n"), $this->getTitle(),
  328.             //    (strlen($data) > 512 ? substr($data, 0, 512) : $data));
  329.             get_current_dpuniverse()->storeTell($this, $data, $binded_env);
  330.             return;
  331.         }
  332.         if (is_array($data)) {
  333.             $data = $data[$this->displayMode];
  334.         }
  335.  
  336.         // Gets the message type from $data which holds a string with the format
  337.         // <type>tellstuff</type>. If no type was given in the $data string, we
  338.         // consider it type 'message':
  339.         list($mtype_start, $data, $mtype_end) = $this->_tellParseTag($data);
  340.  
  341.         $this->_tellDoTell($mtype_start, $data, $mtype_end);
  342.     }

  344.     private function _tellParseTag($data)
  345.     {
  346.          if (strlen($data) >=3 && substr($data, 0, 1) == '<'
  347.                 && FALSE !== ($pos = strpos($data, '>'))) {
  348.             $mtype_start = substr($data, 1, $pos - 1);
  349.             $endpos = strrpos($data, '<');
  350.             $mtype_end = substr($data, $endpos + 2, -1);
  351.  
  352.             $data = "<![CDATA[" . substr($data, strlen($mtype_start) + 2,
  353.                 $endpos - strlen($mtype_start) - 2) . ']]>';
  354.         } else {
  355.             $mtype_start = $mtype_end = 'message';
  356.             $data = "<![CDATA[$data]]>";
  357.         }
  358.  
  359.         return array($mtype_start, $data, $mtype_end);
  360.     }

  362.     private function _tellDoTell($mtype_start, $data, $mtype_end)
  363.     {
  364.         $universe = get_current_dpuniverse();
  365.  
  366.         if (isset($this->_GET['ajax'])) {
  367.             if ($mtype_end === 'message' && $data === '<![CDATA[empty]]>') {
  368.                 $mtype_start = $mtype_end = '';
  369.                 $data = '1';
  370.             }
  371.         }
  372.         if ($data !== '1' || !$universe->isToldSomething()) {
  373.             if ($mtype_start !== '') {
  374.                 $data = "<$mtype_start>$data</$mtype_end>";
  375.             }
  376.             if ($data !== '1') {
  377.                 echo sprintf(dptext("Telling %s: %s\n"), $this->getTitle(),
  378.                     (strlen($data) > 256 ? substr($data, 0, 256) : $data));
  379.             }
  380.             if ($mtype_end === '' || $mtype_end === 'header'
  381.                     || $mtype_end === 'cookie' || $mtype_end === 'location') {
