  107.         /**
  108.          * Sets the message shown to the user when an action fails
  109.          *
  110.          * Call this method from action methods, for example actionFoo(). When the
  111.          * user performs an action but the action fails, there are two ways for the
  112.          * method implementing the action to communicate the failure to the user.
  113.          *
  114.          * <ol>
  115.          * <li>

  122.          * <br>

  124.          * action.<br><br>

  137.          * the action. If it doesn't find any, the previously set failure message

  287.     /**
  288.      * Tries to perform the action given by the living object
  289.      *
  290.      * Called by the system to handle both actions performed by clicking on
  291.      * menus and actions from the input area. The first method will result in an

  297.      * be in another object. Handles failure messages when the action failed.
  298.      *
  299.      * This method can also be used to force living objects to perform actions.
  300.      *
  301.      * @param   string  $action     the action given by the living object
  302.      * @return  boolean TRUE for success, FALSE for unsuccessful action
  303.      * @see     setActionFailure(), getActionFailure(), setActionDefaultFailure(),
  304.      *           getActionDefaultFailure(), DpObject::addAction(),
  305.      *           DpObject::getActions(), DpObject::getActionsMenu(),
  306.      *           DpObject::getTargettedActions(),
  307.      *           DpObject::performActionSubject()
  308.      */
  309.     public function performAction($action)
  310.     {
  311.         global $grCurrentDpObject;
  312.  
  313.         $this->lastActionTime = !isset($this->lastActionTime)
  314.             ? new_dp_property(time()) : time();
  315.  
  316.         $action = trim($action);
  317.         $grCurrentDpObject = $this;
  318.  
  319.         echo sprintf(dptext("Action by %s:%s\n"), $this->getTitle(), $action);
  320.  
  321.         $rval = (bool)$this->performActionSubject($action, $this);
  322.         if (TRUE !== $rval) {
  323.             if (strlen($action) && $action !== dptext('look')
  324.                     && $action !== dptext('l')) {
  325.                 $action_failure = $this->getActionFailure();
  326.                 if (!isset($this->_GET['menuaction'])) {
  327.                     $this->tell($action_failure);
  328.                     if ($this->isUser && $this === get_current_dpuser()) {
  329.                         get_current_dpuniverse()->setToldSomething();
  330.                     }
  331.                 }
  332.                 $this->setActionFailure($this->getActionDefaultFailure());
  333.             }
  334.         }
  335.         $grCurrentDpObject = FALSE;
  336.         return $rval;
  337.     }
