  104.     // First, we create the source image with GD Image

  110.  
  111.     // Translate color codes

  113.     if (preg_match("/^#?($Hex$Hex)($Hex$Hex)($Hex$Hex)$/", $BGColor, $Parts)) {
  114.         $BGColors = array(hexdec($Parts[1]), hexdec($Parts[2]), hexdec($Parts[3]));
  115.     } else {

  117.     }
  118.     if (preg_match("/^#?($Hex$Hex)($Hex$Hex)($Hex$Hex)$/", $TextColor, $Parts)) {
  119.         $TextColors = array(hexdec($Parts[1]), hexdec($Parts[2]),
  120.             hexdec($Parts[3]));
  121.     } else {
  122.         $TextColors = array(0, 0, 0);
  123.     }
  124.  
  125.     // Colors allocations
  126.     $BGColor = imagecolorallocate($ImageRessource, $BGColors[0], $BGColors[1],
  127.     $BGColors[2]);
  128.     $TextColor = imagecolorallocate($ImageRessource, $TextColors[0],
  129.     $TextColors[1], $TextColors[2]);
  130.
