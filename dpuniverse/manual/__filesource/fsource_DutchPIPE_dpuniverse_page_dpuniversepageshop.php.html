  129.     public function actionBuy($verb, $noun)
  130.     {
  131.         $user = get_current_dpuser();
  132.  
  133.         $store = get_current_dpuniverse()->getDpObject(DPUNIVERSE_PAGE_PATH
  134.             . 'store.php');
  135.  
  136.         if (!($item = $store->isPresent($noun))) {
  137.             $user->tell(sprintf(
  138.                 dptext('That item ("%s") is not for sale here.<br />'), $noun));
  139.             return TRUE;
  140.         }
  141.  
  142.         $price = $store->getBuyValue($item->value);
  143.  
  144.         if ($user->credits < $price) {
  145.             $user->tell(sprintf(
  146.                 dptext("%s costs %d credits, which you don't have.<br />"),
  147.                 ucfirst($item->getTitle(DPUNIVERSE_TITLE_TYPE_DEFINITE)),
  148.                 $price));
  149.             return TRUE;
  150.         }
  151.  
  152.         if (FALSE !== ($move_result = $item->moveDpObject($user))) {
  153.             switch ($move_result) {
  154.             case E_MOVEOBJECT_HEAVY:
  155.                 $err = sprintf(
  156.                     dptext("You cannot carry %s, it is too heavy. Drop something else first.<br />"),
  157.                     $item->getTitle(DPUNIVERSE_TITLE_TYPE_DEFINITE));
  158.                 break;
  159.             case E_MOVEOBJECT_VOLUME:
  160.                 $err = sprintf(
  161.                     dptext("You cannot carry %s, it takes too much volume. Drop something else first.<br />"),
  162.                     $item->getTitle(DPUNIVERSE_TITLE_TYPE_DEFINITE));
  163.                 break;
  164.             default:

  166.                     dptext("You fail to buy %s as it cannot be moved.<br />"),
  167.                     $item->getTitle(DPUNIVERSE_TITLE_TYPE_DEFINITE));
  168.             }
  169.  
  170.             $user->tell($err);
  171.             return TRUE;
  172.         }
  173.  
  174.         $user->credits -= $price;
  175.         $user->tell(sprintf(dptext("You buy %s for %d credits.<br />"),
  176.             $item->getTitle(DPUNIVERSE_TITLE_TYPE_INDEFINITE), $price));

  178.             $this->tell(sprintf(dptext("%s buys %s.<br />"),
  179.                 $user->getTitle(DPUNIVERSE_TITLE_TYPE_DEFINITE),

  182.         return TRUE;
  183.     }

  185.     public function actionSell($verb, $noun)
  186.     {
  187.         $user = get_current_dpuser();
  188.  
  189.         $store = get_current_dpuniverse()->getDpObject(DPUNIVERSE_PAGE_PATH
  190.             . 'store.php');
  191.  
  192.         if (!($item = $user->isPresent($noun))) {
  193.             $user->tell(sprintf(
  194.                 dptext("You don't own that item (\"%s\").<br />"), $noun));
  195.             return TRUE;
  196.         }
  197.  
  198.         if ($item->isDrink && $item->isFull) {
  199.             $user->tell(dptext("The shop doesn't accept filled glasses and liquids.<br />"));
  200.             return TRUE;
  201.         }
  202.  
  203.         if (0 == ($price = $store->getSellValue($item->value))) {
  204.             $user->tell(sprintf(
  205.                 dptext("%s has no value.<br />"),
  206.                 $item->getTitle(DPUNIVERSE_TITLE_TYPE_DEFINITE)));
  207.             return TRUE;
  208.         }
  209.         if ($item->isNoStore) {
  210.             $item->removeDpObject();
  211.         } elseif ($item->isNoSell
  212.                 || FALSE !== ($move_result = $item->moveDpObject($store))) {
  213.             $user->tell(sprintf(dptext("%s cannot be sold.<br />"),
  214.                 $item->getTitle(DPUNIVERSE_TITLE_TYPE_DEFINITE)));
  215.             return TRUE;
  216.         }
  217.  
  218.         $user->credits += $price;
  219.         $user->tell(sprintf(dptext("You sell %s for %d credits.<br />"),
  220.             $item->getTitle(DPUNIVERSE_TITLE_TYPE_INDEFINITE), $price));
  221.         if (($env = $user->getEnvironment()) && $env === $this) {
  222.             $this->tell(sprintf(dptext("%s sells %s.<br />"),
  223.                 $user->getTitle(DPUNIVERSE_TITLE_TYPE_DEFINITE),
  224.                 $item->getTitle(DPUNIVERSE_TITLE_TYPE_INDEFINITE)), $user);
  225.         }
  226.         return TRUE;
  227.     }
