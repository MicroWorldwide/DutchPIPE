  406.      * "full".
  407.      *
  408.      * @param   string  $verb       the action, "drink"
  409.      * @param   string  $noun       what to drink, could be empty
  410.      * @return  boolean TRUE for action completed, FALSE otherwise
  411.      */
  412.     public function actionDrink($verb, $noun)
  413.     {
  414.         $user = get_current_dpuser();
  415.         if (empty($noun)) {
  416.             $user->setActionFailure(dptext('What do you want to drink?<br />'));
  417.             return FALSE;
  418.         }
  419.  
  420.         if (FALSE !== ($env = $this->getEnvironment())
  421.                 && $env->isPresent($noun) !== $this) {
  422.             $user->setActionFailure(sprintf(dptext('You can\'t drink: %s<br />'),
  423.                 $noun));
  424.             return FALSE;
  425.         }
  426.  
  427.         if ($env !== $user) {
  428.             $user->tell(dptext('You must pick it up first.<br />'));
  429.             return TRUE;
  430.         }
  431.         if (FALSE === $this->getProperty('is_full')) {
  432.             $user->tell(ucfirst(sprintf(dptext('%s is empty.<br />'),
  433.                 $this->getTitle(DPUNIVERSE_TITLE_TYPE_DEFINITE))));
  434.             return TRUE;
  435.         }
  436.  
  437.         $this->mOrigIds = $this->getIds();
  438.         $this->mOrigTitle = $this->getTitle();
  439.         $this->mOrigTitleImg = $this->getTitleImg();
  440.         $this->mOrigBody = $this->getBody();
  441.  
  442.         if (FALSE !== ($env = $user->getEnvironment())) {
  443.             $env->tell(ucfirst(sprintf(dptext('%s drinks %s.<br />'),
  444.                 $user->getTitle(DPUNIVERSE_TITLE_TYPE_DEFINITE),
  445.                 $this->getTitle(DPUNIVERSE_TITLE_TYPE_INDEFINITE))),
  446.                 $user);
  447.             $env->tell('<window autoclose="2500" styleclass="dpwindow_drink">'
  448.                 . '<h1>' . dptext('BUUUUUUUUUUUUUURRRP!') . '</h1></window>',
  449.                 $user);
  450.         }
  451.  
  452.         $user->tell(sprintf(dptext('You drink %s.<br />'),
  453.             $this->getTitle(DPUNIVERSE_TITLE_TYPE_INDEFINITE)));
  454.         $user->tell('<window autoclose="2500" styleclass="dpwindow_drink">'
  455.             . '<h1>' . dptext('BUUUUUUUUUUUUUURRRP!') . '</h1></window>');
