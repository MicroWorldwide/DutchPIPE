  366.     private function handleLinkdead()
  367.     {
  368.         $cur_time = time();
  369.  
  370.         /* Time the user's browser should have done a page or AJAX request */
  371.         $linkdeath_time = $cur_time - DPUNIVERSE_LINKDEATH_KICKTIME;
  372.  
  373.         /*
  374.          * Currently, this function is triggered by a user http request. If
  375.          * there's one user and he leaves, then if the next user enters, say a
  376.          * minute later (this time is defined here), he should not see
  377.          * 'X leaves the site.'
  378.          */
  379.         $showmsg_time = $linkdeath_time - DPUNIVERSE_LINKDEATH_SHOWMSGTIME;
  380.  
  381.         /*
  382.          * Bots may be visible a bit longer, since they don't have Javacript and
  383.          * sometimes do a lot of requests. With a short 'kick' time, users would
  384.          * get a lot of 'Bot enters the site' and 'Bot leaves the site'
  385.          * messages.
  386.          */
  387.         $botkick_time = $cur_time - DPUNIVERSE_BOT_KICKTIME;
  388.  
  389.         $showbot_time = $botkick_time - DPUNIVERSE_LINKDEATH_SHOWBOTTIME;
  390.  
  391.         foreach ($this->mDpUsers as $i => &$u) {
  392.             /*
  393.              * Throw out people who lost connection or browsed elsewhere. Need
  394.              * to think this one out (move them to a void, etc.), something
  395.              * simple for now.
  396.              */
  397.             $lastrequest_time = $u[_DPUSER_TIME_LASTREQUEST];
  398.             $ajax_capable = $u[_DPUSER_OBJECT]->getProperty(
  399.                 'is_ajax_capable');
  400.  
  401.             /* The "linkdeath" check */
  402.             if (($ajax_capable && $lastrequest_time < $linkdeath_time)
  403.                     || (!$ajax_capable && $lastrequest_time < $botkick_time)) {
  404.  
  405.                 /*
  406.                  * This method is called before an instance of
  407.                  * CurrentDpUserRequest is created and the current http request
  408.                  * is handled. However, the tell() functions operate on the
  409.                  * current request. Therefore, any sound made should be stored
  410.                  * for the next cycle, otherwise the people get wrong messages.
  411.                  * tell() in the User class checks for this variable:
  412.                  */
  413.                 $this->mNoDirectTell = TRUE;
