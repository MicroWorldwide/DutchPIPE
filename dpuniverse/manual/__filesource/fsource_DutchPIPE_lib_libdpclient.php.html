  80.                 SOL_TCP))) {
  81.             $gLastErrorMsg = '<h1>' . DPSERVER_SOCKERR_MSG . '</h1>'

  83.                 "socket_create(): unable to create socket [%u]: %s\n"),
  84.                 socket_last_error(), socket_strerror(socket_last_error()));

  86.         }
  87.         if (FALSE === @socket_connect($socket, DPSERVER_SOCKET_ADDRESS,
  88.                 DPSERVER_SOCKET_PORT)) {
  89.             $gLastErrorMsg = '<h1>' . DPSERVER_SOCKERR_MSG . '</h1>'
  90.                 . sprintf(dptext(
  91.                 "socket_create(): unable to connect [%u]: %s\n"),

  94.         }
  95.     } else {

  99.     }
  100.  
  101.     /* Talk to the DutchPIPE server and write user variables to it */
  102.     if (!isset($_SESSION)) {
  103.         $_SESSION = array();
  104.     }
  105.  
  106.     /* Serialize $_SERVER, $_SESSION, ... variables so they can be sent */
  107.     $in = serialize(array($_SERVER, $_SESSION, $_COOKIE, $_GET, $_POST,
  108.         $_FILES));
  109.     if (TRUE === DPSERVER_BASE64_CLIENT2SERVER) {
  110.         $in = base64_encode($in);
  111.     }
  112.     $in = "<vars>$in</vars>\r\nquit\r\n";
  113.  
  114.     socket_write($socket, $in, strlen($in));
  115.  
  116.     /* Read and process server reply, filter header info given by the server,
  117.       put the remainder in $output */
  118.     $cookie_set = $remove_guest_cookie = $remove_registered_cookie = FALSE;
  119.  
  120.     for ($output = ''; $buf = socket_read($socket, DPSERVER_CLIENT_CHUNK); ) {
  121.         $output .= $buf;
  122.     }
  123.  
  124.     $arroutput = explode(chr(0), $output);
  125.     $output = '';
  126.     foreach ($arroutput as $buf) {
  127.         if (!strlen($buf) || isset($newlocation)) {
  128.             continue;
  129.         }
  130.         $bufdec = TRUE === DPSERVER_BASE64_SERVER2CLIENT ? base64_decode($buf)
  131.             : $buf;
  132.         if (strlen($bufdec) > 11 && substr($bufdec, 0, 11) == "Set-Login: ") {
  133.             $cookie_data = substr($bufdec, 11);
  134.             setcookie(DPSERVER_COOKIE_NAME, $cookie_data, FALSE, '/');
  135.             $cookie_set = TRUE;
  136.         }
  137.  
  138.         elseif (strlen($bufdec) > 17 && FALSE !== ($pos1 = strpos($bufdec,
  139.                 "<header><![CDATA[")) && FALSE !== ($pos2 = strpos($bufdec,
  140.                 "]]></header>")) && $pos2 > $pos1 + 12) {
  141.             $header_data = substr($bufdec, 0, $pos2);
  142.             $header_data = substr($header_data, $pos1 + 17);
  143.             header($header_data);
  144.         }
  145.  
  146.         elseif (strlen($bufdec) > 17 && FALSE !== ($pos1 = strpos($bufdec,
  147.                 "<cookie><![CDATA[")) && FALSE !== ($pos2 = strpos($bufdec,
  148.                 "]]></cookie>")) && $pos2 > $pos1 + 12) {
  149.             $cookie_data = substr($bufdec, 0, $pos2);
  150.             $cookie_data = substr($cookie_data, $pos1 + 17);
  151.             if ($cookie_data == 'removeguest') {
  152.                 $remove_guest_cookie = TRUE;
  153.             }
  154.             elseif ($cookie_data == 'removeregistered') {
  155.                 $remove_registered_cookie = TRUE;
  156.             }
  157.             else {
  158.                 setcookie(DPSERVER_COOKIE_NAME, $cookie_data, time() + 630720000, '/');
  159.                 $cookie_set = TRUE;
  160.             }
  161.         }
  162.  
  163.         elseif (strlen($bufdec) > 19 && FALSE !== ($pos1 = strpos($bufdec,
  164.                 "<location><![CDATA[")) && FALSE !== ($pos2 = strpos($bufdec,
  165.                 "]]></location>")) && $pos2 > $pos1 + 14) {
  166.             $bufdec = substr($bufdec, 0, $pos2);
  167.             $bufdec = substr($bufdec, $pos1 + 19);
  168.             $newlocation = $bufdec;
  169.             $output = "<location><![CDATA[$bufdec]]></location>";
  170.             continue;
  171.         }
  172.         else {
  173.             $output .= TRUE === DPSERVER_BASE64_SERVER2CLIENT ? base64_decode($buf) : $buf;
  174.         }
  175.     }
  176.  
  177.     /* Close cocket, return server reply */
  178.     socket_close($socket);
  179.  
  180.     if (FALSE === $cookie_set) {
  181.         if (FALSE !== $remove_guest_cookie) {
  182.             setcookie(DPSERVER_COOKIE_NAME, FALSE, FALSE, '/');
  183.         }
  184.         if (FALSE !== $remove_registered_cookie) {
  185.             setcookie(DPSERVER_COOKIE_NAME, FALSE, time() - 3600, '/');
  186.         }
  187.     }
  188.  
  189.     if (isset($newlocation)) {
  190.         $newlocation = $newlocation == '/' ? '/'
  191.             : DPSERVER_CLIENT_URL . "?location=$newlocation";
  192.         if (!isset($_GET) || !isset($_GET['ajax'])) {
  193.             header("Location: $newlocation");
  194.             exit;
  195.         }
  196.     }
  197.  
  198.     return $output;
  199. }

  231. /**
  232.  * Sets cookies transmitted with the xml we got from the server

  236. function handle_cookies(&$xml)
  237. {
  238.     foreach ($xml->cookie as $id => $cookie) {
  239.         foreach($xml->cookie->attributes() as $attname => $attval) {
  240.             switch ($attname) {
  241.             case 'name':
  242.                 $name = $attval;
  243.                 break;
  244.             case 'expire':
  245.                 $expire = strlen($attval) ? $attval : FALSE;
  246.                 break;
  247.             case 'path':
  248.                 $path = $attval;
  249.                 break;
  250.             case 'domain':
  251.                 $domain = $attval;
  252.                 break;
  253.             case 'secure':
  254.                 $secure = $attval;
  255.                 break;
  256.             default:
  257.                 break;
  258.             }
  259.         }
  260.  
  261.         if (!isset($name)) {
  262.             break;
  263.         }
  264.  
  265.         if (!isset($secure)) {
  266.             if (!isset($domain)) {
  267.                 if (!isset($path)) {
  268.                     if (!isset($expire)) {
  269.                         setcookie($name, $cookie, FALSE, '/');
  270.                         $cookie_debug = "setcookie($name, $cookie)";
  271.                     } else {

  273.                         $cookie_debug = "setcookie($name, $cookie, $expire)";
  274.                     }
  275.                 } else {
  276.                     setcookie($name, $cookie,
  277.                         (!isset($expire) ? FALSE : $expire), $path);
  278.                     $cookie_debug = "setcookie($name, $cookie, "
  279.                         . (!isset($expire) ? 'FALSE' : $expire) . ", $path)";
  280.                 }
  281.             } else {
  282.                 setcookie($name, $cookie, (!isset($expire) ? FALSE : $expire),
  283.                     (!isset($path) ? '' : $path), $domain);
  284.                 $cookie_debug = "setcookie($name, $cookie, " . (!isset($expire)
  285.                     ? 'FALSE' : $expire) . ", " . (!isset($path) ? '' : $path)
  286.                     .  ", $domain)";
  287.             }
  288.         } else {
  289.             setcookie($name, $cookie, (!isset($expire) ? FALSE : $expire),
  290.                 (!isset($path) ? '' : $path), (!isset($domain) ? '' : $domain),
  291.                 $secure);
  292.             $cookie_debug = "setcookie($name, $cookie, " . (!isset($expire) ?

  299. /**

  369.         $body = '<h1>' . dptext('Error fetching page. Invalid page XML.')
  370.             . '</h1>';
  371.     }
  372.  
  373.     if (strlen($dpelements)) {

  381.             </script>\n";
  382.     }
  383.  
  384.     $inputtopmargin = !sizeof($messages) ? '0' : '10';
  385.     $windows = implode("\n", $windows);
  386.     $messages = implode("\n", $messages);
  387.  
  388.     /* Otherwise serve the page with the retrieved content in it */
  389.     echo "<?xml version=\"1.0\" encoding=\"utf-8\"?>\n";
  390.     if (strlen($html)) {
  391.         $html = str_replace(
  392.             array('{$dpelements}', '{$windows}', '{$messages}',
  393.                 '{$inputtopmargin}'),
  394.             array($dpelements, $windows, $messages, $inputtopmargin), $html);
  395.         echo $html;
  396.     } else {
  397.         require_once(DPSERVER_TEMPLATE_PATH . DPSERVER_TEMPLATE_FILE);
  398.     }
  399. }
