  105. function inherit($path)
  106. {
  107.     /* PHP 5.1.5 kludge symlink bug fix */
  108.     $files = get_included_files();
  109.     $pos = strrpos($path, '/');
  110.     if (FALSE !== $pos && $pos != strlen($path) -1) {
  111.         $file = substr($path, $pos + 1);
  112.     }
  113.     foreach ($files as $f) {
  114.         $pos = strrpos($f, '/');
  115.         if (FALSE !== $pos && $pos != strlen($f) -1) {
  116.             if ($file === substr($f, $pos + 1)) {
  117.                 return;
  118.             }
  119.         }
  120.     }
  121.  
  122.     require_once(DPUNIVERSE_PREFIX_PATH . $path);
  123. }

  154. /**
  155.  * Gets a short, descriptive string for an age in seconds
  156.  *
  157.  * This method is used for example to show the age of users. A new user could
  158.  * just be "1 minute and 15 seconds" old, while addicts could be hanging around
  159.  * for "2 years, 74 days and 20 hours".
  160.  *
  161.  * @param      int       $age        number of seconds
  162.  * @return     string    Short, descriptive age string
  163.  */
  164. function get_age2string($age)
  165. {
  166.     $rest_age= $age;
  167.  
  168.     $rval = array();
  169.     if ($age >= 31536000) {
  170.         $tmp = floor($age / 31536000);
  171.         $rval[] = 1 === $tmp ? dptext('1 year') : dptext('%d years', $tmp);
  172.         $rest_age = $age % 31536000;
  173.     }
  174.     if ($rest_age >= 86400) {
  175.         $tmp = floor($rest_age / 86400);
  176.         $rval[] = 1 === $tmp ? dptext('1 day') : dptext('%d days', $tmp);
  177.         $rest_age = $rest_age % 86400;
  178.     }
  179.     if ($rest_age >= 3600) {
  180.         $tmp = floor($rest_age / 3600);
  181.         $rval[] = 1 === $tmp ? dptext('1 hour') : dptext('%d hours', $tmp);
  182.         $rest_age = $rest_age % 3600;
  183.     }
  184.  
  185.     if ($age < 31536000) {
  186.         if ($rest_age >= 60) {
  187.             $tmp = floor($rest_age / 60);
  188.             $rval[] = 1 === $tmp ? dptext('1 minute')
  189.                 : dptext('%d minutes', $tmp);
  190.             $rest_age = $rest_age % 60;
  191.         }
  192.         if ($age < 86400) {
  193.             $rval[] = 1 === $rest_age ? dptext('1 second')
  194.                 : dptext('%d seconds', $rest_age);
  195.         }
  196.     }
  197.  
  198.     if (1 === ($sz = count($rval))) {
  199.         return $rval[0];
  200.     }
  201.  
  202.     $last = $rval[$sz - 1];
  203.     $rval = array_slice($rval, 0, -1);
  204.  
  205.     return implode(', ', $rval) . ' ' . dptext('and') . ' ' . $last;
  206. }
