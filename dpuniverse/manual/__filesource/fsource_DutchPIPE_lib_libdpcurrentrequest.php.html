  241.     /**
  242.      * Finds if user is already on this site, otherwise creates a user object
  243.      *
  244.      * Tries to find cookie info that goes with the HTTP request to link the
  245.      * request to a user object already on this site. Otherwise, creates either
  246.      * a new guest or registered user object in the universe. Registered users
  247.      * should have a cookie set in their browser with login info.
  248.      *
  249.      * Returns true if a user object was found or created, or FALSE in case of
  250.      * an error.
  251.      *
  252.      * @return  boolean TRUE for user object found or created, FALSE otherwise
  253.      */
  254.     private function _findOrCreateDpUser()
  255.     {
  256.         if (TRUE === $this->_findCookieData() &&
  257.                 FALSE !== ($u = $this->mrDpUniverse->findCurrentDpUser(
  258.                 $this->mCookieId, $this->mCookiePass))) {
  259.             /* This user is already on this site */
  260.             $this->_initExistingDpUser($u);
  261.             return TRUE;
  262.         }
  263.  
  264.         if (isset($this->__GET) && isset($this->__GET['ajax']) &&
  265.                 (!isset($this->__GET['standalone']) ||
  266.                 (isset($this->__GET['seq']) && (int)$this->__GET['seq'] > 0))) {
  267.             /*
  268.              * There's an AJAX request coming from someone without cookies.
  269.              * The DutchPIPE cookie should have been set in the page request
  270.              * which comes first.
  271.              */
  272.             $this->mCookieId = FALSE;
  273.             //echo dptext("NO COOKIE\n");
  274.             return FALSE;
  275.         }
  276.         /*
  277.          * A page request from someone entering the site. Initialize data
  278.          * for either a registered user (by obtaining a cookie) or a guest
  279.          * user.
  280.          */
  281.         if (FALSE === $this->_findAndInitRegisteredDpUser() &&
  282.                 FALSE === $this->_initDpGuest()) {
  283.             return FALSE;
  284.         }
  285.  
  286.         /* Create the new user object with the initialized data */
  287.         $this->_createDpUser();
  288.         return TRUE;
  289.     }

  291.     /**
  292.      * Checks if the current connection can be tied to a user.
  293.      *
  294.      * Checks and possibly creates user and location objects, handles passing of
  295.      * variables such server, cookie and get arrays.
  296.      *
  297.      * @return  boolean TRUE for valid user request, FALSE otherwise
  298.      */
  299.     private function _findCookieData()
  300.     {
  301.         if (isset($this->__COOKIE)
  302.                 && isset($this->__COOKIE[DPSERVER_COOKIE_NAME])
  303.                 && strlen($this->__COOKIE[DPSERVER_COOKIE_NAME])
  304.                 && strlen($cookie_data =
  305.                     trim($this->__COOKIE[DPSERVER_COOKIE_NAME], ';'))
  306.                 && sizeof($cookie_data = explode(';', $cookie_data))
  307.                 && 2 === sizeof($cookie_data)) {
  308.             $this->mCookieId = $cookie_data[0];
  309.             $this->mCookiePass = $cookie_data[1];
  310.  
  311.             return strlen($this->mCookieId) && strlen($this->mCookiePass);
  312.         }
  313.         return FALSE;
  314.     }

  334.     /**
  335.      * Finds and initializes user data for a registered user entering the site
  336.      *
  337.      * If cookie data was sent by the browser and it is valid data for a
  338.      * registered user, TRUE is returned, otherwise FALSE is returned.
  339.      *
  340.      * @return  boolean TRUE for request from registered user, FALSE otherwise
  341.      */
  342.     private function _findAndInitRegisteredDpUser()
  343.     {
  344.         /*
  345.          * A registered user. Skip Ajax database check for now, but what
  346.          * are the security implications?
  347.          */
  348.  
  349.         if (!isset($this->mCookieId) || !isset($this->mCookiePass)) {
  350.             return FALSE;
  351.         }
  352.  
  353.         $result = mysql_query("
  354.             SELECT
  355.                 userUsername,userAvatarNr,userDisplayMode,
  356.                 userEventPeopleEntering,userEventPeopleLeaving,
  357.                 userEventBotsEntering
  358.             FROM
  359.                 Users
  360.             WHERE
  361.                 userCookieId='" . addslashes($this->mCookieId) . "'
  362.             AND
  363.                 userCookiePassword='" . addslashes($this->mCookiePass) . "'
  364.             ");
  365.  
  366.         if (empty($result) || !($row = mysql_fetch_array($result))) {
  367.             return FALSE;
  368.         }
  369.  
  370.         $this->mUsername = $row[0];
  371.         $this->mUserAvatarNr = $row[1];
  372.         $this->mUserDisplayMode = $row[2];
  373.         $alert_events = array();
  374.         if ('1' === $row[3]) {
  375.             $alert_events['people_entering'] = TRUE;
  376.         }
  377.         if ('1' === $row[4]) {
  378.             $alert_events['people_leaving'] = TRUE;
  379.         }
  380.         if ('1' === $row[5]) {
  381.             $alert_events['bots_entering'] = TRUE;
  382.         }
  383.         if (count($alert_events)) {
  384.             $this->mUserAlertEvents = $alert_events;
  385.         }
  386.  
  387.         $this->mIsRegistered = TRUE;
  388.  
  389.         return TRUE;
  390.     }
  391.     /**
  392.      * Initializes guest variables, sets cookie
  393.      *
  394.      * Sets up a random Guest#X user name and sends a cookie to the guest to
  395.      * store user information.
  396.      *
  397.      * Some experimental stuff is going on with search bots, so you can see
  398.      * them crawling the site. This will later be turned into an option.
  399.      *
  400.      * FALSE is returned if the user's client didn't report a "user agent"
  401.      * string. This is mandatory. The guest member variables will not be set up.
  402.      *
  403.      * @return  boolean TRUE for succesful setup, FALSE otherwise
  404.      */
  405.     private function _initDpGuest()
  406.     {
  407.         $this->mIsKnownBot = $this->_findAgentAndKnownBots();
  408.  
  409.         if (FALSE === $this->mUserAgent) {
  410.             return FALSE;
  411.         }
  412.  
  413.         if (FALSE === $this->mIsKnownBot) {
  414.             $this->mUsername = sprintf(dptext('Guest#%d'),
  415.                 $this->mrDpUniverse->getGuestCnt());
  416.         } else {
  417.             $this->mUsername = $this->mIsKnownBot;
  418.         }
  419.  
  420.         $this->mCookieId = make_random_id();
  421.         $this->mCookiePass = make_random_id();
  422.         $this->mIsRegistered = FALSE;
  423.  
  424.         //echo "Set-Login: {$this->mCookieId};{$this->mCookiePass}\n";
  425.         $this->mrDpUniverse->tellCurrentDpUserRequest(
  426.             "Set-Login: {$this->mCookieId};{$this->mCookiePass}");
  427.         return TRUE;
  428.     }

  430.     /**
  431.      * Finds if the HTTP request came from a known search bot
  432.      *
  433.      * Every browser sends a "user agent" string, DutchPIPE stores some of the
  434.      * more well-known user agent strings for search bots in the database, and
  435.      * assigns them a special avatar when they enter the site.
  436.      *
  437.      * :WARNING: Browsers/spiders/bots that do not report a user agent, are
  438.      * denied access.
  439.      *
  440.      * @return  string|boolean Avatar title for known bots, FALSE otherwise
  441.      */
  442.     private function _findAgentAndKnownBots()
  443.     {

  636.     /**

  640.      */
  641.     private function _isValidLocation()
  642.     {
  643.         //echo dptext("_isValidLocation: checking location \"%s\"\n",
  644.         //    !isset($this->__GET['location']) ? '' : $this->__GET['location']);
  645.         if (isset($this->__GET['location'])
  646.                 && !strlen($this->__GET['location'])) {
  647.             return TRUE;
  648.         }
  649.  
  650.         if (!isset($this->__GET['location'])) {
  651.             return isset($this->__GET['getdivs']);
  652.         }
  653.  
  654.         if (FALSE !== strpos($this->__GET['location'], '..')) {
  655.             return FALSE;
  656.         }
  657.  
  658.         if (FALSE !== ($pos = strpos($this->__GET['location'], '?'))) {
  659.             $this->__GET['location'] =
  660.                 substr($this->__GET['location'], 0, $pos);
  661.         }
  662.  
  663.         if (FALSE !== ($pos = strpos($this->__GET['location'], '#'))) {
  664.             $this->__GET['location'] =
  665.                 substr($this->__GET['location'], 0, $pos);
  666.         }
  667.  
  668.         /* Experimental */
  669.         if (isset($this->__GET['proxy'])
  670.                 || 0 === strpos($this->__GET['location'], '/mailman2/')) {
  671.             return TRUE;
  672.         }
  673.  
  674.         if (($len = strlen(DPSERVER_HOST_URL)) < strlen($this->__GET['location'])
  675.                 && DPSERVER_HOST_URL === substr($this->__GET['location'], 0,
  676.                 $len)) {
  677.             return TRUE;
  678.         }
  679.         return (file_exists(DPUNIVERSE_PREFIX_PATH . $this->__GET['location'])
  680.                 && is_file(DPUNIVERSE_PREFIX_PATH . $this->__GET['location']))
  681.             || (file_exists(DPUNIVERSE_WWW_PATH . $this->__GET['location'])
  682.                 && is_file(DPUNIVERSE_WWW_PATH . $this->__GET['location']));
  683.     }

  839.             }
